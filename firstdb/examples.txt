1. Select all rows
SELECT * FROM people;

2. Select specific columns
SELECT f_name, l_name FROM people;

3. Filter rows with WHERE
SELECT * FROM people WHERE result > 50;

4. Pattern matching with LIKE
SELECT * FROM people WHERE f_name LIKE 'J%';

5. Sort results with ORDER BY
SELECT f_name, result FROM people ORDER BY result DESC;

6. Remove duplicates with DISTINCT
SELECT DISTINCT l_name FROM people;

7. Count all rows and return in a new column
SELECT COUNT(*) AS total_people FROM people;
ex: count the number of people called Snow
SELECT COUNT(*) AS snow FROM people where l_name = 'Snow';

8. Count rows matching a condition
SELECT COUNT(*) AS passed FROM people WHERE result >= 40;
ex: How many scored between 34 and 70 inclusive
SELECT COUNT(*) AS range FROM people WHERE result >= 34 and result <= 70;


9. Group rows with GROUP BY
SELECT l_name, COUNT(*) AS total FROM people GROUP BY l_name;
ex: return a result that shows the grouped results. i.e. how many of each result is there
SELECT result, COUNT(*) AS results FROM people GROUP BY result;

10. Aggregates per group
SELECT l_name, COUNT(*) AS num_people, ROUND(AVG(result),2) AS avg_result, MAX(result) AS max_result FROM people GROUP BY l_name;
ex: Output the mean of all results to 2 DP and the max mark in the class
SELECT ROUND(AVG(result),2) AS mean, MAX(result) AS best FROM people; 

11. Limit the number of rows
SELECT f_name, result FROM people ORDER BY result DESC LIMIT 3;

12. Filter groups with HAVING
Break this one down showing the order in which each is evaluated as it has everything. Change 4 to 1 to show it working
SELECT l_name, COUNT(*) AS total FROM people WHERE result > 20 GROUP BY l_name HAVING COUNT(*) > 1 ORDER BY total DESC LIMIT 4;


13. Update rows
UPDATE people SET result = result + 5 WHERE f_name = 'John' AND l_name = 'Snow';
ex: Give everyone an extra 5 marks. Then take them back off again.
UPDATE people SET result = result + 5;

14. Delete rows
DELETE FROM people WHERE result < 20;

15. Column aliases - put in "" for spaces
SELECT f_name AS first, l_name AS last, result AS score FROM people;

16. BETWEEN range filter
SELECT * FROM people WHERE result BETWEEN 30 AND 70;

17. IN list filter
SELECT * FROM people WHERE l_name IN ('Snow', 'Jet');

18. ORDER BY multiple columns
SELECT * FROM people ORDER BY l_name ASC, result DESC;

19. Can use functions (non-standard). e.g. average. These are not SQL standard
SELECT UPPER(l_name), result FROM people WHERE result >=70; 
SELECT CURDATE();

20. Can use regex to find stuff more then say %eng% in the like clause. '.' means any char
SELECT * FROM people WHERE f_name REGEXP '^j';
SELECT * FROM people WHERE f_name REGEXP '^jo.n';
